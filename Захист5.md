def factorial(n):
    if n > 0:
        return n * factorial(n - 1)
    elif n==0:
        return 1
    elif n < 0:
        return n * factorial(n + 1)

def main():
    m = int(input())
    n = int(input())
    print(factorial(m)/factorial(m+n))

if __name__ == '__main__':
    print(factorial(-3))
